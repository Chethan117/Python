AND NOT EXISTS (
    SELECT 1
    FROM authzn c
    WHERE c.account_id = b.account_id
      AND c.card_number = b.card_number
      AND FLOOR(MONTHS_BETWEEN(c.authorisation_request_process_date, b.authorisation_request_process_date)) BETWEEN 1 AND 11
      AND c.authorisation_request_process_date < a.authorisation_request_process_date
      AND c.authorization_amount BETWEEN 0.9 * b.authorization_amount AND 1.1 * b.authorization_amount
      AND c.authorization_response_code IN (1,2,3)
      AND c.pos_entry_mthd_cd IN (01,10,81)
)
